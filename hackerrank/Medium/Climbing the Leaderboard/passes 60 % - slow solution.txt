def climbingLeaderboard(ranked, player):
    # Write your code here
    
    for score in player:
        game_ranks = list(ranked)
        
        game_ranks.append(score)
        
        # we use yield here to create a generator object as a return for this function
        yield sorted(set(game_ranks), reverse=True).index(score) + 1
